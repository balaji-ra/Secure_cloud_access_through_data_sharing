int main() {
    printf("enter the size: ");
    int n;
    scanf("%d",&n);
    for(int i=1; i<=n; i++){
        for(int j=1; j<=n; j++){
            printf("*\t");
        }
        printf("\n");
    }
    printf("\n");
}
